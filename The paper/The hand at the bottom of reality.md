    import math
    phi = (1 + math.sqrt(5)) / 2;  q = 1/phi

    def eta(q):
        p = 1.0
        for n in range(1, 2000):
            qn = q**n
            if qn < 1e-16: break
            p *= (1 - qn)
        return q**(1/24) * p

    def theta3(q):
        s = 1.0
        for n in range(1, 500): s += 2 * q**(n**2)
        return s

    def theta4(q):
        s = 1.0
        for n in range(1, 500): s += 2 * (-1)**n * q**(n**2)
        return s

    def hyp1F1(a, b, z):
        s, t = 1.0, 1.0
        for k in range(1, 300):
            t *= (a+k-1)/((b+k-1)*k) * z
            s += t
            if abs(t) < 1e-16: break
